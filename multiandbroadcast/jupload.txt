package multicastserver;

import java.io.*;
import java.net.*;
/**
 *
 * @author Edwin Babu
 */
public class MulticastServer {


    /**
     * @param args the command line arguments
     */
    public static void main(String[] args) {
        // TODO code application logic here
        DatagramSocket serversocket;
        DatagramPacket packet;
        int port=8888;
        InetAddress addr;
        
        try{
            addr=InetAddress.getByName("224.0.0.3");
            serversocket=new DatagramSocket();
            int i=0;
            String msg;
            System.out.println("Enter any text to send:");           
            while(true)
            {
                BufferedReader in=new BufferedReader(new InputStreamReader(System.in));
                msg=in.readLine();
                packet=new DatagramPacket(msg.getBytes(),msg.getBytes().length,addr,port);
                serversocket.send(packet);
                System.out.println("Server: "+ msg);
                Thread.sleep(500);
                i++;
            }
                
        }
        catch(Exception e)
        {
            System.out.println(e);
        }
    }
    
}

package multicastclient;
import java.io.*;
import java.net.*;
/**
 *
 * @author Edwin Babu
 */
public class MulticastClient {

    /**
     * @param args the command line arguments
     */
    public static void main(String[] args) {
        // TODO code application logic here
        InetAddress addr;
        MulticastSocket clientsocket;
        DatagramPacket packet;
        byte[] buf=new byte[256];
        int port=8888;
        String msg;
        try
        {
            
            addr=InetAddress.getByName("224.0.0.3");
            clientsocket=new MulticastSocket(port);
            clientsocket.joinGroup(addr);
            while(true)
            {
                packet=new DatagramPacket(buf,buf.length);
                clientsocket.receive(packet);
                msg=new String(buf,0,buf.length);
                System.out.println("Socket 1 recieved msg: "+msg);
            }
        }
        catch(Exception e)
        {
            System.out.println(e);
        }
    }
    
}
